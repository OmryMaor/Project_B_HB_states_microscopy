import numpy as np
from itertools import product
from qutip import *
import matplotlib.pyplot as plt
import matplotlib.colors as mcolors
from matplotlib.gridspec import GridSpec
import pandas as pd


def generate_combinations(K, N):
    """
    Generate configurations where the sum of particles in K signal modes
    is less than or equal to N. The reference mode is implicit.
    """
    all_combinations = product(range(N + 1), repeat=K)
    valid_combinations = [comb for comb in all_combinations if sum(comb) <= N]
    return valid_combinations

def diagonalize(rho):
    """Diagonalize the density matrix rho"""
    eigenvalues, eigenvectors = rho.eigenstates()
    return eigenvalues, eigenvectors

def compute_rho_derivative(rho, vals, vecs, generators, a):
    """Computes the matrix elements of the derivative ∂_a(ρ) = -1j[n_a, ρ] in the eigenbasis."""
    n_a = generators[a]
    U = Qobj(np.hstack([v.full() for v in vecs]), dims=[rho.dims[0], rho.dims[0]])
    n_a_eigen = (U.dag() * n_a * U).full()
    diffs = vals[np.newaxis, :] - vals[:, np.newaxis]
    return -1j * diffs * n_a_eigen

def calculate_QFIM(rho, vals, vecs, generators):
    """Implements the Quantum Fisher Information Matrix (QFIM) using the spectral representation."""
    dim = len(vals)
    K = len(generators)
    qfim = np.zeros((K, K))
    
    derivatives_eigen = []
    for a in range(K):
        derivatives_eigen.append(compute_rho_derivative(rho, vals, vecs, generators, a))

    for a in range(K):
        for b in range(a, K):
            term_sum = 0
            for n in range(dim):
                for m in range(dim):
                    v_sum = vals[n] + vals[m]
                    if v_sum > 1e-14:
                        elem_a = derivatives_eigen[a][n, m]
                        elem_b = derivatives_eigen[b][m, n]
                        term_sum += (2.0 / v_sum) * np.real(elem_a * elem_b)
            qfim[a, b] = term_sum
            qfim[b, a] = term_sum
    return qfim

def N_free(initial_signal_state, eta_vec, N, K):
    """Independent Noise Channel via Stinespring dilation"""
    D = N + 1
    V_total = tensor([qeye(D)] * (2 * K))
    
    for i in range(K):
        theta_i = np.arccos(np.sqrt(eta_vec[i]))
        ops_s = [qeye(D)] * (2 * K); ops_s[i] = destroy(D)
        ops_e = [qeye(D)] * (2 * K); ops_e[K + i] = destroy(D)
        a_s = tensor(ops_s)
        a_e = tensor(ops_e)
        h_int = a_s.dag() * a_e - a_s * a_e.dag()
        V_total = (theta_i * h_int).expm() * V_total

    env_vacuum = tensor([basis(D, 0)] * K)
    rho_joint = ket2dm(tensor(initial_signal_state, env_vacuum))
    rho_final_joint = V_total * rho_joint * V_total.dag()
    return rho_final_joint.ptrace(list(range(K)))

def N_int(initial_signal_state, eta_vec, N, K):
    """N_int (Shared Environment beam splitter)"""
    D = N + 1
    a_e = tensor([qeye(D)] * K + [destroy(D)])
    h_tot = tensor([qeye(D)] * (K + 1)) * 0.0
    
    for i in range(K):
        theta_i = np.arccos(np.sqrt(eta_vec[i]))
        ops_s = [qeye(D)] * (K + 1); ops_s[i] = destroy(D)
        a_s = tensor(ops_s)
        h_int_i = a_s.dag() * a_e - a_s * a_e.dag()
        h_tot += theta_i * h_int_i

    V_total = h_tot.expm()
    rho_joint = ket2dm(tensor(initial_signal_state, basis(D, 0)))
    rho_final = V_total * rho_joint * V_total.dag()
    return rho_final.ptrace(list(range(K)))

def N_tot(initial_ket, eta_vec, p, N, K):
    """Implements the total noisy channel"""
    rho_free = N_free(initial_ket, eta_vec, N, K)
    rho_int = N_int(initial_ket, eta_vec, N, K)
    rho_tot = p * rho_free + (1.0 - p) * rho_int
    return rho_tot


def initialize_HB_state(N, d, D=None):
    """
    Generate the Holland-Burnett State HB(n, d) mathematically exactly as generalized via Multiport QFT.
    n photons input per mode. n = N / (d+1).
    Total modes = d+1 (Mode 0: Reference, Modes 1..d: Signal).
    """
    D = N + 1 if D is None else D
    modes = d + 1
    n = N // modes
    if N % modes != 0:
        raise ValueError(f"For HB states, N={N} must be a multiple of d+1={modes}")
        
    omega = np.exp(2j * np.pi / modes)
    
    a_ops = []
    for i in range(modes):
        ops = [qeye(D)] * modes
        ops[i] = create(D)
        a_ops.append(tensor(ops))
        
    b_ops = []
    for k in range(modes):
        b_k = 0
        for j in range(modes):
            b_k += (1.0 / np.sqrt(modes)) * (omega ** (j * k)) * a_ops[j]
        b_ops.append(b_k)
        
    state = tensor([basis(D, 0)] * modes)
    for k in range(modes):
        for _ in range(n):
            state = b_ops[k] * state

    state = state.unit() 
            
    combinations = generate_combinations(d, N)
    basis_kets_K = []
    for comb in combinations:
        state_modes_K = [basis(D, n_signal) for n_signal in comb]
        basis_kets_K.append(tensor(state_modes_K))

    coeffs = []
    for comb in combinations:
        n_0 = N - sum(comb)
        target_ket = tensor([basis(D, n_0)] + [basis(D, n_i) for n_i in comb])
        amplitude = target_ket.overlap(state)
        coeffs.append(amplitude)
        
    return np.array(coeffs), basis_kets_K, combinations

def initialize_humphreys_state(N, d, D=None):
    """
    Eq. 8, achieving the bound in Eq. 10.
    """
    D = N + 1 if D is None else D
    alpha = 1.0 / np.sqrt(d + np.sqrt(d))
    beta = 1.0 / np.sqrt(1 + np.sqrt(d))
    
    combinations = generate_combinations(d, N)
    basis_kets_K = []
    for comb in combinations:
        basis_kets_K.append(tensor([basis(D, n) for n in comb]))
        
    coeffs = []
    for comb in combinations:
        if tuple(comb) == tuple([0] * d):
            coeffs.append(beta)
        elif sum(comb) == N and max(comb) == N:
            coeffs.append(alpha)
        else:
            coeffs.append(0.0)
            
    return np.array(coeffs), basis_kets_K, combinations

def construct_ket(coeffs, basis_kets):
    ket = sum(coeffs[i] * basis_kets[i] for i in range(len(coeffs)))
    return ket.unit()


def evaluate_state_metrics(initial_ket, N, K, eta_vec, p):
    """Evaluate QFI and Total Variance for a specific state under noise"""
    rho_out = N_tot(initial_ket, eta_vec, p, N, K)
    vals, vecs = diagonalize(rho_out)
    
    D = N + 1
    generators = []
    for i in range(K):
        ops = [qeye(D)] * K
        ops[i] = num(D)
        generators.append(tensor(ops))
        
    qfim = calculate_QFIM(rho_out, vals, vecs, generators)
    max_qfi = np.trace(qfim)
    total_var = np.trace(np.linalg.pinv(qfim)) if np.linalg.det(qfim) != 0 else np.inf
    return max_qfi, total_var

def evaluate_grid(N, d, state_coeffs, basis_kets, eta_values, p_values):
    """Runs a grid of evaluations and returns a DataFrame"""
    results = []
    initial_ket = construct_ket(state_coeffs, basis_kets)
    for eta in eta_values:
        for p in p_values:
            qfi, crb = evaluate_state_metrics(initial_ket, N, d, [eta]*d, p)
            results.append({"eta": eta, "p": p, "max_qfi": qfi, "total_var": crb})
            print(f"Computed eta={eta}, p={p:.1f} | QFI={qfi:.3f}, CRB={crb:.3f}")
    return pd.DataFrame(results)

def plot_comparison(df, N, d, title_prefix="Holland-Burnett State Analysis"):
    """
    Reproduces the exact style of the original cells 21 & 23.
    """
    plt.rcParams.update({'figure.facecolor': 'white', 'axes.facecolor': 'white'})
    
    etas = df['eta'].unique()
    for eta in etas:
        subset = df[df['eta'] == eta].sort_values(by='p')
        fig = plt.figure(figsize=(9.0, 9.0))
        gs = GridSpec(2, 1, height_ratios=[4, 1], hspace=0.45)
        ax_qfi = fig.add_subplot(gs[0])
        ax_var = fig.add_subplot(gs[1])
        
        sql_val = (d**2) / N
        noon_val = (d**3) / (N**2)
        optimal_hl_val = (d * (1 + np.sqrt(d))**2) / (4 * (N**2))
        
        best_crb = subset['total_var'].min()
        
        lhs_bound = min(optimal_hl_val, best_crb)
        rhs_bound = max(sql_val, best_crb)
        margin = max((rhs_bound - lhs_bound) * 0.1, 0.1)
        
        ax_var.set_xlim(lhs_bound - margin, rhs_bound + margin)
        ax_var.set_ylim(-1.0, 1.0)
        
        bar_height = 0.15
        ax_var.fill_between([lhs_bound, rhs_bound], -bar_height, bar_height, color='#E0E6ED', zorder=1)
        
        benchmarks = [optimal_hl_val, noon_val, sql_val]
        benchmark_labels = [
            f"$\\mathbf{{Humphreys Bound: \\approx {optimal_hl_val:.3f}}}$",
            f"$\\mathbf{{Heisenberg: d^3/N^2 \\approx {noon_val:.2f}}}$",
            f"$\\mathbf{{SQL: d^2/N \\approx {sql_val:.2f}}}$"
        ]
        
        for i, (val, label) in enumerate(zip(benchmarks, benchmark_labels)):
            ax_var.plot([val, val], [-bar_height, bar_height], color='black', linewidth=2.5, zorder=2)
            if i == 0: ax_var.text(val, bar_height + 0.1, label, color='black', fontsize=11, ha='left', va='bottom')
            elif i == 2: ax_var.text(val, bar_height + 0.1, label, color='black', fontsize=11, ha='right', va='bottom')
            else: ax_var.text(val, bar_height + 0.1, label, color='black', fontsize=11, ha='center', va='bottom')
        
        colors = plt.cm.viridis(np.linspace(0, 1, len(subset)))
        
        for i, (idx, row) in enumerate(subset.iterrows()):
            ax_qfi.scatter(row['p'], row['max_qfi'], s=120, label=f"p={row['p']:.1f}", color=colors[i], edgecolors='black', zorder=5)
            if np.isclose(row['p'], 0.0):
                ax_var.plot([row['total_var'], row['total_var']], [-bar_height, bar_height], color=colors[i], linewidth=8, solid_capstyle='butt', zorder=5)
                
        ax_var.plot([best_crb, best_crb], [-bar_height, bar_height], color='darkorange', linewidth=4, linestyle='--', zorder=6)
        ax_var.text(best_crb, -bar_height - 0.1, f"Best State CRB: {best_crb:.3f}", color='darkorange', fontsize=11, ha='center', va='top', weight='bold')
        
        hl_qfi_val = (d**2) / noon_val
        multipixel_qfi_val = (d**2) / optimal_hl_val
        
        ax_qfi.axhline(y=hl_qfi_val, color='red', linestyle='--', linewidth=2, zorder=2)
        ax_qfi.text(0.01, hl_qfi_val + 0.05, f"Heisenberg Limit QFI ($\\approx {hl_qfi_val:.2f}$)", color='red', fontsize=10, ha='left', va='bottom', weight='bold')
        
        ax_qfi.axhline(y=multipixel_qfi_val, color='dodgerblue', linestyle='-.', linewidth=2, zorder=2)
        ax_qfi.text(0.01, multipixel_qfi_val + 0.05, f"Multipixel QFI ($\\approx {multipixel_qfi_val:.2f}$)", color='dodgerblue', fontsize=10, ha='left', va='bottom', weight='bold')
        
        ax_qfi.set_xlabel("Mixture Parameter $p$ (0 = Full Interaction, 1 = No Interaction)", fontsize=11)
        ax_qfi.set_ylabel("Maximized $\\mathrm{Tr}(\\mathcal{I}_{\\theta})$", fontsize=11)
        ax_qfi.grid(True, linestyle=':', alpha=0.4)
        
        ax_var.set_yticks([]); ax_var.spines['left'].set_visible(False)
        ax_var.spines['right'].set_visible(False); ax_var.spines['top'].set_visible(False)
        ax_var.spines['bottom'].set_visible(False); ax_var.set_xticks([])
        ax_var.set_title("Cramer-Rao Bound (CRB) Analysis", fontsize=13, pad=5)
        
        handles, labels = ax_qfi.get_legend_handles_labels()
        plt.subplots_adjust(left=0.15, bottom=0.35, right=0.95, top=0.95)
        fig.legend(handles, labels, title=f"{title_prefix} ($\\eta={eta}$)", loc='upper center', bbox_to_anchor=(0.5, 0.33), fontsize=9, frameon=True, ncol=6)
        
        plt.show()

def plot_fidelity_heatmap(df_metrics, fidelity_val, title=""):
    """
    Reproduces cell 24/25 heatmap style mapping fidelity with the Humphreys state across p and eta grids.
    For an fixed exact pure state, the input fidelity is a constant scalar.
    We just demonstrate the footprint. Realistically, we replace 'fidelities_sq' with Total Variance heatmaps 
    or just label the static fidelity clearly!
    """
    # Simple CRB heatmap to replace the "fidelity" optimization sweep mapping, 
    # since fidelity is constant for our fixed states!
    p_vals = sorted(df_metrics['p'].unique())
    eta_vals = sorted(df_metrics['eta'].unique())
    crb_matrix = np.zeros((len(eta_vals), len(p_vals)))
    
    for i, e in enumerate(eta_vals):
        for j, pv in enumerate(p_vals):
            crb_matrix[i, j] = df_metrics[(df_metrics['eta'] == e) & (df_metrics['p'] == pv)]['total_var'].values[0]
            
    fig, ax = plt.subplots(figsize=(10, 8))
    plt.rcParams['axes.facecolor'] = 'white'
    
    sc = ax.imshow(crb_matrix, cmap='RdYlBu', origin='lower', aspect='auto')
    
    ax.set_yticks(range(len(eta_vals)))
    ax.set_yticklabels([f"{e:.2f}" for e in eta_vals])
    ax.set_xticks(range(len(p_vals)))
    ax.set_xticklabels([f"{pv:.1f}" for pv in p_vals])
    
    ax.set_ylabel("Transmission Parameter $\\eta$", fontsize=15, labelpad=10)
    ax.set_xlabel("Mixture Parameter $p$", fontsize=15, labelpad=10)
    
    cbar = plt.colorbar(sc, pad=0.04)
    cbar.set_label("Total Variance (CRB)", fontsize=12)
    
    ax.set_title(title, fontsize=16, pad=20)
    plt.grid(True, linestyle=':', alpha=0.4)
    plt.tight_layout()
    plt.show()


# == 1. Variables Definition ==
N_photons = 3
d_phases = 2

# == 2. State Construction ==
hb_coeffs, basis_kets, combs = initialize_HB_state(N_photons, d_phases)
hump_coeffs, _, _ = initialize_humphreys_state(N_photons, d_phases)

ket_hb = construct_ket(hb_coeffs, basis_kets)
ket_hump = construct_ket(hump_coeffs, basis_kets)

print("----- Holland-Burnett State Initialization -----")
for c, cb in zip(hb_coeffs, combs):
    if np.abs(c) > 1e-5: print(f"{c:.3f} | {cb} > (Signal modes)")

print("\n----- Humphreys Reference State Initialization -----")
for c, cb in zip(hump_coeffs, combs):
    if np.abs(c) > 1e-5: print(f"{c:.3f} | {cb} > (Signal modes)")

# Pure state fidelity between HB and Humphreys Reference:
fid = np.abs(ket_hb.overlap(ket_hump))**2
print(f"\nInitial State Fidelity |<ψ_HB|ψ_s>|^2 : {fid:.4f}\n")

# == 3. Evaluation on Grid ==
# For plotting, we evaluate against mixture 'p' [0 to 1] and specific etas
p_grid = np.round(np.linspace(0.0, 1.0, 11), 1)
eta_grid = [0.8, 1.0]

print("Evaluating HB State Matrix Profile...")
df_hb = evaluate_grid(N_photons, d_phases, hb_coeffs, basis_kets, eta_grid, p_grid)

# == 4. Generate Reproductions of Required Plots ==
# 4A. Plotting QFI/CRB analysis as specified by Cells 21, 23 (The dual axis scatter + bounded line logic)
plot_comparison(df_hb, N_photons, d_phases, title_prefix="Holland-Burnett State Analysis")

# 4B. The Cell 24/25 Mapping alternative (Heatmap substituting Fidelity array with CRB footprint!)
plot_fidelity_heatmap(df_hb, fid, title=f"HB State Performance: QFI CRB Profiling (Initial Fidelity {fid:.2f})")
